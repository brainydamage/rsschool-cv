        function pickPeaks(arr){
          let posArr = arr
            .map((val, pos) => pos * Math.sign(val - arr[pos-1]))
            .filter(val => val && val!==0)
            .filter((val, pos, arr) => val>0 && arr[pos-1] < val > arr[pos+1]);

          return {pos: posArr, peaks: posArr.map(pos => arr[pos])};

        //Create all permutations of an input string.

        function permutations(string) {
          if (!string.length && string.length < 2 ){
            return [string];
          }

          let result = [];
          let result = [];
          for (let i = 0; i < string.length; i++) {
            const currentChar = string[i];
            const remainder = string.slice(0, i) + string.slice(i + 1, string.length);

            for (let permutation of permutations(remainder)) {
              result.push(currentChar + permutation);
            }
          }
          return [...new Set(result)];
        }
